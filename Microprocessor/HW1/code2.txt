ORG 00H
MOV DPTR,#300H

CHECK:
MOV A,P2
MOV R5,#08H
MOV R6,#00H
CALL COUNT
CALL PRINT
SJMP CHECK

COUNT:
JB ACC.0,ROTATE
INC R6

ROTATE:
RR A
DJNZ R5,COUNT
RET

PRINT:
MOV A,R6
MOVC A,@A+DPTR
MOV P1,A
CALL DELAY

DELAY:
MOV R3,#250

HERE:
DJNZ R3,HERE
RET
ORG 300H

TABLE:
DB 11000000B, 11111001B, 10100100B,10110000B,10011001B,10010010B,10000011B,11011000B,10000000B;0,1,2,3,4,5,6,7,8
END
