ORG 00H
MAIN:
MOV A,#0FEH
SW:
JNB P2.1,SW1L
SW1H:
JB P2.0,STOP ;11 안눌린것
SJMP RIGHT ;10 sw0 눌림
SW1L:
JB P2.0,LEFT ;01 sw1 눌림
SJMP OFF ;00모두눌린것
OFF:
MOV A,#0FFH
MOV P1,A
SJMP MAIN
STOP:
MOV P1,A
SJMP SW
RIGHT:
RR A
MOV P1,A
CALL DELAY
SJMP SW
LEFT:
RL A
MOV P1,A
CALL DELAY
SJMP SW
DELAY:
MOV R2,#200
AGAIN:
MOV R3,#250
HERE:
DJNZ R3,HERE
DJNZ R2,AGAIN
RET
END
